-- use case 4: many cancels in a row
add_type("nuts")
add_type("bolts")
add_type("hammers")  -- report in alphabetical order
add_product("bolts", 100)
add_product("nuts", 1005)
add_product("hammers", 191000000)
add_order(<<["nuts", 29], ["bolts", 31]>>)
add_order(<<["hammers", 22], ["nuts", 5]>>) 
add_order(<<["nuts", 100]>>)   
add_order(<<["nuts", 10], ["hammers", 91000000]>>)
cancel_order(2)
cancel_order(4)
cancel_order(3)
cancel_order(1)
add_order(<<["nuts", 29], ["bolts", 31]>>)
cancel_order(5)
add_order(<<["hammers", 22], ["nuts", 5]>>) 
add_order(<<["nuts", 100]>>)   
add_order(<<["nuts", 10], ["hammers", 91000000]>>)
